CREATE TABLE  E_BOOK (CALLNO VARCHAR2(4000) NOT NULL PRIMARY KEY, 
	NAME VARCHAR2(4000), 
	AUTHOR VARCHAR2(4000), 
	PUBLISHER VARCHAR2(4000), 
	QUANTITY NUMBER, 
	ISSUED NUMBER);

CREATE TABLE  E_LIBRARIAN (ID NUMBER NOT NULL PRIMARY KEY,
	NAME VARCHAR2(4000), 
	PASSWORD VARCHAR2(4000), 
	EMAIL VARCHAR2(4000), 
	MOBILE NUMBER);


CREATE TABLE  E_ISSUEBOOK (CALLNO VARCHAR2(4000) NOT NULL , 
	STUDENTID VARCHAR2(4000) NOT NULL, 
	STUDENTNAME VARCHAR2(4000), 
	STUDENTMOBILE NUMBER, 
	ISSUEDDATE DATE, 
	RETURNSTATUS VARCHAR2(4000) );
